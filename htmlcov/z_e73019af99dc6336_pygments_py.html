17def _get_styles(formatter: HtmlFormatter, prefix: str) -> None: 

18 """Get styles out of a formatter, where everything has the correct prefix.""" 

19 for line in formatter.get_linenos_style_defs(): 

20 yield f"{prefix} {line}" 

21 yield from formatter.get_background_style_defs(prefix)

25def get_pygments_stylesheet(light_style: str, dark_style: str) -> str: 

26 """Generate the theme-specific pygments.css. 

27 

28 There is no way to tell Sphinx how the theme handles modes. 

29 """ 

30 light_formatter = HtmlFormatter(style=light_style) 

31 dark_formatter = HtmlFormatter(style=dark_style) 

32 

33 lines = [] 

34 

35 light_prefix = 'html[data-theme="light"] .highlight' 

36 lines.extend(_get_styles(light_formatter, prefix=light_prefix)) 

37 

38 dark_prefix = 'html[data-theme="dark"] .highlight' 

39 lines.extend(_get_styles(dark_formatter, prefix=dark_prefix)) 

40 

41 return "\n".join(lines)

44def overwrite_pygments_css(app: Sphinx, exception=None): 

45 """Overwrite pygments.css to allow dynamic light/dark switching. 

46 

47 Sphinx natively supports config variables `pygments_style` and 

48 `pygments_dark_style`. However, quoting from 

49 www.sphinx-doc.org/en/master/development/theming.html#creating-themes 

50 

51 The pygments_dark_style setting [...is used] when the CSS media query 

52 (prefers-color-scheme: dark) evaluates to true. 

53 

54 This does not allow for dynamic switching by the user, so at build time we 

55 overwrite the pygment.css file so that it embeds 2 versions: 

56 

57 - the light theme prefixed with "[data-theme="light"]" 

58 - the dark theme prefixed with "[data-theme="dark"]" 

59 

60 Fallbacks are defined in this function in case the user-requested (or our 

61 theme-specified) pygments theme is not available. 

62 """ 

63 if exception is not None: 

64 return
